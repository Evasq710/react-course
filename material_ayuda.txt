===== COMANDOS ======

# Init project
npm create vite@latest

# Axios Installation
npm i axios

# React Router in DATA MODE | DECLARATIVE MODE (FRAMEWORK MODE requires different installation)
npm i react-router

# Tailwind CSS Installation
npm i tailwindcss @tailwindcss/vite

# shadcn Installation (a set of components for React that uses a CODE DISTRIBUTION SYSTEM; it uses Tailwind CSS)
npm i -D @types/node
# npx > node package execute ...
npx shadcn@latest init 
# Adding components to project
npx shadcn@latest add

# Sonner: a toast component for React
npm install sonner

# Zod for TypeScrfipt SCHEMA VALIDATION 
npm install zod

# para pruebas unitarias (-D == --save-dev)
npm i -D vitest

# para pruebas de componentes de React
npm install --save-dev @testing-library/react @testing-library/dom @types/react @types/react-dom

# Un s√≥lo comando para configurar entorno de pruebas
npm install --save-dev @testing-library/react @testing-library/dom vitest jsdom 

# To mock axios actions while testing
npm i axios-mock-adapter --save-dev

# When the project is built, we can run a static HTTP server 
# Installation:
sudo npm install --global http-server
# Launch a server inside our dist foler:
http-server -o
